CREATE TABLE READERS (
   ID INT PRIMARY KEY,
   NAME VARCHAR(8)
);

CREATE TABLE BLOGS_READERS (
   R_ID INT,
   B_ID INT,
   PRIMARY KEY (R_ID, B_ID),
   FOREIGN KEY (R_ID) REFERENCES READERS(ID),
   FOREIGN KEY (B_ID) REFERENCES BLOGS(ID)
);

CREATE TABLE BLOGS (
   ID INT PRIMARY KEY,
   TITLE VARCHAR(50),
   DESCRIPTION VARCHAR(4000)
);
